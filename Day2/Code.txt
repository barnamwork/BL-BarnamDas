public class Problem12 {
    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        // Input base and height (in inches)
        System.out.print("Enter base in inches: ");
        double base = input.nextDouble();

        System.out.print("Enter height in inches: ");
        double height = input.nextDouble();

        // Area in square inches
        double areaInSqInches = 0.5 * base * height;

        // Conversion: 1 inch = 2.54 cm
        double areaInSqCm = areaInSqInches * 2.54 * 2.54;

        // Output
        System.out.printf(
            "Area of the triangle is %.2f square inches and %.2f square centimeters%n",
            areaInSqInches,
            areaInSqCm
        );

        input.close();
    }
}

        String viaCity = input.nextLine();

        System.out.print("Enter to city: ");
        String toCity = input.nextLine();

        System.out.print("Enter distance from source to via city (miles): ");
        double fromToVia = input.nextDouble();

        double viaToFinalCity = input.nextDouble();

        System.out.print("Enter time taken (hours): ");
        double timeTaken = input.nextDouble();

        double totalDistance = fromToVia + viaToFinalCity;
        double speed = totalDistance / timeTaken;

            "%s travelled from %s to %s via %s covering %.2f miles in %.2f hours with an average speed of %.2f mph%n",

            fromCity,
            toCity,
            viaCity,
            totalDistance,
            timeTaken,
            speed
        );

        input.close();
    }
}

public class Problem11 {
    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        // Input values
        System.out.print("Enter Principal: ");
        double principal = input.nextDouble();

        System.out.print("Enter Rate of Interest: ");
        double rate = input.nextDouble();

        System.out.print("Enter Time (in years): ");
        double time = input.nextDouble();

        // Simple Interest calculation

            principal,
            rate,

        );

    }
}
